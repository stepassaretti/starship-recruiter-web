        const rd = RACES[raceKey];


        let str = rand(rd.str[0], rd.str[1]);

        let int = rand(rd.int[0], rd.int[1]);

        let cha = rand(rd.cha[0], rd.cha[1]);

        let hate = rand(0, 100);

        let name = NAMES[rand(0, NAMES.length - 1)];


        // Enforce Biological Traits to not generate Saboteurs by default

        if (raceKey === "Sugorian") hate = rand(0, 50);

        if (raceKey === "Blorg") hate = rand(0, 20);

        if (raceKey === "Sky-Elf") { int = Math.max(str, cha) + rand(1, 10); }

        if (raceKey === "Human") { while (str === int && int === cha) { str++; } }

        if (raceKey === "Beetlegunner") { str = Math.max(str, cha + 1); }

        if (raceKey === "Giant") { str = int + cha + rand(1, 10); }

        if (raceKey === "Droid") { name += "-" + rand(1, 99); }

        if (raceKey === "Robomantoid") { hate = rand(20, 100); }

        if (raceKey === "Barbitulian") { int = cha; }

        if (raceKey === "Cheesponge" || raceKey === "Aquasim") { str = Math.min(int, cha) - rand(1, 5); if (str < 1) str = 1; }

        if (raceKey === "Space-Cat") { cha = Math.max(str, int) + rand(1, 10); }

        if (raceKey === "Velkhorian") { hate = rand(0, 60); }

        if (raceKey === "Fungi" || raceKey === "Drakhen") { hate = rand(80, 100); }

        if (raceKey === "Drakhen") { int = str; }

        if (raceKey === "Void-Demon") { cha = str + rand(1, 10); hate = rand(0, 10); }

        if (raceKey === "Oak-Person") { cha = rand(1, 4); }

        if (raceKey === "Long-Neck" || raceKey === "Squid") { int = str + rand(1, 10); }

        if (raceKey === "Sodio") { str = Math.max(str, int + 1); }


        const safeRaceName = raceKey.toLowerCase();

        const maxImgs = rd.images || 12;

        const portraitVariant = rand(1, maxImgs);


        return {

            id: id,

            name: name,

            race: raceKey,

            str: str, int: int, cha: cha, hate: hate,

            exp: ROLES[rand(0, ROLES.length - 1)],

            portrait: `images/${safeRaceName}_portrait_${portraitVariant}.png`,

            cost: Math.floor((str + int + cha) * 1.5) + rand(5, 15)

        };

    }


    function generateMission() {

        const types = [

            { id: "ASSAULT", stat: ["str"], t: 1 },

            { id: "DIPLOMACY", stat: ["cha"], t: 1 },

            { id: "RESEARCH", stat: ["int"], t: 1 },

            { id: "RESCUE", stat: ["str", "int"], t: 2 },

            { id: "SMUGGLING", stat: ["int", "cha"], t: 2 },

            { id: "EVACUATION", stat: ["str", "int", "cha"], t: 3 }

        ];

        

        let available = types.filter(t => t.t === 1);

        if (state.day > 4) available = available.concat(types.filter(t => t.t === 2));

        if (state.day > 8) available = available.concat(types.filter(t => t.t === 3));


        const m = available[rand(0, available.length - 1)];

        

        // STEEPER DIFFICULTY SCALING

        const targetBase = 45 + Math.floor(state.day * 7.5); 


        let targets = {};

        m.stat.forEach(s => targets[s] = targetBase + rand(-5, 5));


        return {

            title: m.id + " OP-" + rand(100, 999),

            type: m.id,

            reqs: m.stat,

            targets: targets,

            // STEEPER MIN SUCCESS SCALING (Starts at 66, goes up to 95)

            minReq: Math.min(95, 65 + Math.floor(state.day * 1.2)),

            reward: 500 + (state.day * 90)

        };

    }


    function updateShip() {

        if (state.day <= 3) state.ship = { name: "Speeder", max: 5, min: 3 };

        else if (state.day <= 6) state.ship = { name: "Scout", max: 6, min: 4 };

        else if (state.day <= 10) state.ship = { name: "Gunship", max: 7, min: 4 };

        else state.ship = { name: "Destroyer", max: 8, min: 5 };


        state.roster = new Array(state.ship.max).fill(null);

    }

        updateShip();

        state.mission = generateMission();

        

        const poolSize = 6 + Math.floor(state.day / 2);

        for(let i=0; i<poolSize; i++) {

    }

    let activeCard = null;

    let startX = 0, startY = 0;

    let placeholder = null;

    let originalParent = null;

    let hasMoved = false;


    function bindDragEvents() {

        const wrappers = document.querySelectorAll('.drag-wrapper');


        wrappers.forEach(card => {

            card.addEventListener('pointerdown', (e) => {

                if (e.pointerType === 'mouse' && e.button !== 0) return;

                

                activeCard = card;

                originalParent = card.parentElement;

                hasMoved = false;

                

                const rect = card.getBoundingClientRect();

                startX = e.clientX - rect.left;

                startY = e.clientY - rect.top;


                placeholder = document.createElement('div');

                placeholder.className = 'card-placeholder pixel-staircase';

                originalParent.appendChild(placeholder);


                card.style.width = rect.width + 'px';

                card.style.height = rect.height + 'px';

                card.style.left = rect.left + 'px';

                card.style.top = rect.top + 'px';

                card.style.position = 'fixed';

                card.style.zIndex = '1000';

                card.classList.add('dragging');

                

                card.setPointerCapture(e.pointerId);

                e.preventDefault();

            });

        });

    }


    document.addEventListener('pointermove', (e) => {

        if (!activeCard) return;

        hasMoved = true;

        

        activeCard.style.left = (e.clientX - startX) + 'px';

        activeCard.style.top = (e.clientY - startY) + 'px';


        const slots = document.querySelectorAll('.target-slot');

        let hoveringPanel = false;


        // Check if cursor is over the right panel at all

        const rightPanel = document.getElementById('right-panel');

        if (rightPanel) {

            const rpRect = rightPanel.getBoundingClientRect();

            if (e.clientX > rpRect.left && e.clientX < rpRect.right && e.clientY > rpRect.top && e.clientY < rpRect.bottom) {

                hoveringPanel = true;

            }

        }


        // Clean up all slots first

        slots.forEach(slot => {

            slot.classList.remove('drag-over');

            // If hovering directly over a slot (might be outside the panel bounds technically), also count it

            if (isOverSlot(e.clientX, e.clientY, slot)) {

                hoveringPanel = true;

            }

        });


        // If hovering anywhere in the right panel zone, highlight the FIRST available slot

        if (hoveringPanel) {

            const firstEmptyIndex = state.roster.findIndex(a => a === null);

            if (firstEmptyIndex !== -1) {

                const targetSlotEl = document.querySelector(`.target-slot[data-slot-id="${firstEmptyIndex}"]`);

                if (targetSlotEl) {

                    targetSlotEl.classList.add('drag-over');

                }

            }

        }

    });


    document.addEventListener('pointerup', (e) => {

        if (!activeCard) return;


        const card = activeCard;

        card.releasePointerCapture(e.pointerId);

        card.classList.remove('dragging');

        

        const slots = document.querySelectorAll('.target-slot');

        slots.forEach(s => s.classList.remove('drag-over'));


        const appId = parseInt(card.dataset.id);


        if (!hasMoved) {

            if(placeholder) placeholder.remove();

            resetCardStyles(card);


            const firstEmptySlot = state.roster.findIndex(a => a === null);

            if (firstEmptySlot !== -1) {

                assignToSlot(appId, firstEmptySlot);

            } else {

                updateUI();

            }

            activeCard = null;

            return;

        }


        card.style.transition = 'top 0.15s cubic-bezier(0.2,0,0.3,1), left 0.15s cubic-bezier(0.2,0,0.3,1)';


        let targetX, targetY;

        let targetSlotId = null;


        let droppedInPanel = false;

        

        // Check drop over right panel

        const rightPanel = document.getElementById('right-panel');

        if (rightPanel) {

            const rpRect = rightPanel.getBoundingClientRect();

            if (e.clientX > rpRect.left && e.clientX < rpRect.right && e.clientY > rpRect.top && e.clientY < rpRect.bottom) {

                droppedInPanel = true;

            }

        }

        

        // Check direct slot drop just in case

        for(let slot of slots) {

            if (isOverSlot(e.clientX, e.clientY, slot)) {

                droppedInPanel = true;

                break;

            }

        }


        // If dropped anywhere in the valid zone, snap to the NEXT AVAILABLE empty slot

        if (droppedInPanel) {

            const firstEmptyIndex = state.roster.findIndex(a => a === null);

            if (firstEmptyIndex !== -1) {

                targetSlotId = firstEmptyIndex;

                const targetSlotEl = document.querySelector(`.target-slot[data-slot-id="${firstEmptyIndex}"]`);

                if (targetSlotEl) {

                    const sRect = targetSlotEl.getBoundingClientRect();

                    targetX = sRect.left + (sRect.width - card.offsetWidth) / 2;

                    targetY = sRect.top + (sRect.height - card.offsetHeight) / 2;

                }

            }

        }


        // If it wasn't dropped in the panel OR the panel is completely full

        if (targetSlotId === null) {

            const phRect = placeholder.getBoundingClientRect();

            targetX = phRect.left;

            targetY = phRect.top;

        }


        card.style.left = targetX + 'px';

        card.style.top = targetY + 'px';


        setTimeout(() => {

            if(placeholder) placeholder.remove();

            resetCardStyles(card);

            

            if (targetSlotId !== null) {

                assignToSlot(appId, parseInt(targetSlotId));

            } else {

                updateUI();

            }

            activeCard = null;

        }, 150);

    });


    function resetCardStyles(card) {

        card.style.transition = 'none';

        card.style.position = 'relative';

        card.style.left = ''; 

        card.style.top = '';

        card.style.width = ''; 

        card.style.height = '';

        card.style.zIndex = '10';

    }


    function isOverSlot(x, y, slot) {

        const rect = slot.getBoundingClientRect();

        return (x > rect.left - 20 && x < rect.right + 20 && y > rect.top - 20 && y < rect.bottom + 20);

    }


    function assignToSlot(appId, slotId) {

        const app = state.pool.find(a => a.id === appId);

        if (!app) return;

        

        const currentOccupant = state.roster[slotId];

        const totalCost = state.roster.reduce((sum, a) => sum + (a ? a.cost : 0), 0);

        const replacementCost = currentOccupant ? currentOccupant.cost : 0;

        

        if (state.money < (totalCost - replacementCost + app.cost)) {

            alert("INSUFFICIENT FUNDS. You cannot afford this applicant.");

            updateUI();

            return;

        }


        state.roster[slotId] = app;

        updateUI();

    }


    function returnToPool(slotId) {

        state.roster[slotId] = null;

        updateUI();

    }

    function launchMission() {

        const hired = state.roster.filter(a => a !== null);

        if (hired.length === 0) return;


        const cost = hired.reduce((sum, app) => sum + app.cost, 0);

        state.money -= cost;


        let logs = [];

        let score = 0;


        // 1. RAW SKILL

        let statScores = [];

        state.mission.reqs.forEach(stat => {

            let sum = hired.reduce((acc, app) => acc + app[stat], 0);

            let pct = (sum / state.mission.targets[stat]) * 100;

            statScores.push(pct);

        });

        

        let rawScore = statScores.reduce((a,b)=>a+b, 0) / statScores.length;

        logs.push(`Base Power Output: ${Math.floor(rawScore)}%`);


        // 2. SHIP EFFICIENCY

        let effPct = hired.length / state.ship.min;

        let effMod = 0;

        if (hired.length === state.ship.max) { effMod = 5; logs.push(`Ship at Max Capacity: +5% Bonus`); }

        else if (effPct >= 1) { effMod = 0; logs.push(`Ship Minimum Crew Met: 0% Penalty`); }

        else if (effPct >= 0.7) { effMod = -20; logs.push(`Understaffed (Tier 1): -20% Penalty`); }

        else if (effPct >= 0.4) { effMod = -40; logs.push(`Understaffed (Tier 2): -40% Penalty`); }

        else if (effPct >= 0.15) { effMod = -60; logs.push(`Skeleton Crew (Tier 3): -60% Penalty`); }

        else { effMod = -70; logs.push(`Fatal Understaffing (Tier 4): -70% Penalty`); }


        // 3. CONFLICTS (Mutual Dominance Math)

        let conflictPenalty = 0;

        let conflictCount = 0;


        for(let i=0; i<hired.length; i++) {

            for(let j=i+1; j<hired.length; j++) {

                let a = hired[i]; let rA = RACES[a.race];

                let b = hired[j]; let rB = RACES[b.race];


                let aHatesB = rA.hates.includes(b.race) || rA.hates.includes("ALL");

                let bHatesA = rB.hates.includes(a.race) || rB.hates.includes("ALL");


                if (aHatesB && bHatesA) {

                    // Mutual Conflict! Compare 2 out of 3 stats

                    let aWins = 0, bWins = 0;

                    if(a.str > b.str) aWins++; else if(b.str > a.str) bWins++;

                    if(a.int > b.int) aWins++; else if(b.int > a.int) bWins++;

                    if(a.cha > b.cha) aWins++; else if(b.cha > a.cha) bWins++;


                    if (aWins > bWins) {

                        let pen = 10 + (a.hate/4); conflictPenalty += pen; conflictCount++;

                        logs.push(`MUTUAL: ${a.race} dominated ${b.race}. -${Math.floor(pen)}%`);

                    } else if (bWins > aWins) {

                        let pen = 10 + (b.hate/4); conflictPenalty += pen; conflictCount++;

                        logs.push(`MUTUAL: ${b.race} dominated ${a.race}. -${Math.floor(pen)}%`);

                    } else {

                        let penA = (10 + (a.hate/4))/2;

                        let penB = (10 + (b.hate/4))/2;

                        conflictPenalty += (penA + penB); conflictCount++;

                        logs.push(`MUTUAL DRAW: Bloody stalemate. -${Math.floor(penA+penB)}%`);

                    }

                } else if (aHatesB) {

                    let pen = 10 + (a.hate/4); conflictPenalty += pen; conflictCount++;

                    logs.push(`CONFLICT: ${a.race} attacked ${b.race}. -${Math.floor(pen)}%`);

                } else if (bHatesA) {

                    let pen = 10 + (b.hate/4); conflictPenalty += pen; conflictCount++;

                    logs.push(`CONFLICT: ${b.race} attacked ${a.race}. -${Math.floor(pen)}%`);

                }

            }

        }


        // 4. SYNERGY

        let synergy = 0;

        if (hired.length >= 3 && conflictCount === 0) {

            synergy = 10;

            logs.push(`SYNERGY BONUS: Perfect harmony. +10%`);

        }


        // CALC FINAL

        score = Math.floor(rawScore + effMod - conflictPenalty + synergy);

        

        let overperformance = 0;

        if (score > 100) {

            overperformance = Math.min(20, score - 100);

            score = 100;

        }

        if (score < 0) score = 0;


        // 5. PAYOUT VS FINE

        let earned = 0;

        let isSuccess = false;

        

        if (score >= state.mission.minReq) {

            isSuccess = true;

            earned = Math.floor(state.mission.reward * (score/100));

            if (overperformance > 0) {

                let bonus = Math.floor(state.mission.reward * (overperformance/100));

                earned += bonus;

                logs.push(`OVERPERFORMANCE BONUS: +${bonus} CR`);

            }

        } else {

            let diff = (state.mission.minReq - score) / 100;

            earned = -Math.floor(state.mission.reward * diff);

            logs.push(`FINE APPLIED: Mission requirement (${state.mission.minReq}%) not met.`);

        }


        state.money += earned;


        // Show Modal

        document.getElementById('res-score').innerText = score + "%";

        document.getElementById('res-msg').innerText = isSuccess ? "MISSION SUCCESS" : "MISSION FAILED";

        

        let logHTML = logs.map(l => `<div>> ${l}</div>`).join('');

        logHTML += `<br><div>> NET FINANCIAL CHANGE: ${earned >= 0 ? '+'+earned : earned} CR</div>`;

        document.getElementById('res-log').innerHTML = logHTML;


        document.getElementById('result-overlay').style.display = 'flex';


        if (state.money < 0) state.gameOver = true;

    }


    function nextDay() {

                document.getElementById('modal-btn').innerText = "CONTINUE";

                

                // 3. Restore original button function

                // 4. Hide modal and restart

                document.getElementById('result-overlay').style.display = 'none';

                startDay();

            };

            

            document.getElementById('result-overlay').style.display = 'flex';

        } else {

            state.day++;

            startDay();

        }

    }

    }
